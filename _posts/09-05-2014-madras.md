<script>
var paper = Raphael(0,50,1000,1000);

var Madras = function(originX, originY, widthX, widthY) {
    this.originX = originX;
    this.originY = originY;
    this.widthX = widthX;
    this.widthY = widthY;    
}

Madras.prototype.init = function() {
	var stripe = new Stripe();
	stripe.makeStripe("XL-V", "#f40f7c");
	stripe.makeStripe("XS-H", "#f40f7c");
	stripe.makeStripe("L-H", "#00ab89");
	stripe.makeStripe("M-H", "#fff");
	stripe.makeStripe("S-H", "#ffff00")
}

var Stripe = function() {
	this.offsetX = 0;
	this.offsetY = 0;
	this.startX = 0;
	this.startY = 0;
	this.Xcollection = [];
	this.Ycollection = [];
}

Stripe.prototype.makeStripe = function (readWidth, color) {
	var threads, orientation;
	var threadsLookup = [
		{
			"name" : "XS",
			"number": 10
		},
		{
			"name" : "S",
			"number": 20
		},
		{
			"name" : "M",
			"number" : 25
		},
		{
			"name" : "L",
			"number" : 50
		},
		{
			"name" : "XL",
			"number" : 100
		}
	]
	var cache = readWidth.split("-");
	orientation = cache[1] === "V" ? "vertical" : "horizontal";
	for(var i = 0; i < threadsLookup.length; i++) {
		if (threadsLookup[i].name === cache[0]) {
			threads = threadsLookup[i].number;
		}
	}
	var offsetX = 0;
	var offsetY = 0;
	
	if (orientation === "horizontal") {
		this.Xcollection.push(threads * 3);
		this.startX++;
		
		for (var k = 1; k < this.Xcollection.length; k++) {
			if (this.Xcollection.length === 1) {
				offsetX = 0;
			} else {

				offsetX += this.Xcollection[k-1];
			}
		}
	} else if (orientation === "vertical") {
		this.Ycollection.push(threads * 3);
		this.startY++;
		
		for (var k = 0; k < this.Ycollection.length; k++) {
			if (this.Ycollection.length === 1) {
				offsetY = 0;
			} else {
				offsetY += this.Ycollection[k-1];
			}
		}
	}
	var make;
    for (var j = 0; j<threads; j++) {
        if (j%2 === 0) {
            make = orientation === "vertical" ? new Thread(orientation, 1000, offsetX + 0, offsetY + (j*3)) : new Thread(orientation, 1000, offsetX + (j*3), offsetY +  0);
        } else {
            make = orientation === "vertical" ? new Thread(orientation, 1000, offsetX + 2, offsetY + (j*3)) : new Thread(orientation, 1000, offsetX + (j*3), offsetY +  2);   
        }
		make.defaultDash(color);
    }
}

var Thread = function(direction, length, originX, originY) {
    this.direction = direction;
    this.length = length;
    this.originX = originX;
    this.originY = originY;
    this.multiplier = 4;
}

Thread.prototype.point = function (x, y) {
    return x + "," + y + ",";
}

Thread.prototype.defaultDash = function (color) {
    var x = this.originX;
    var y = this.originY;
    var tr; 
    if (this.direction === "vertical") {
        tr = this.line(this.point(x, y) + this.point(x+this.length, y));
    }
    
    if (this.direction === "horizontal") {
        tr = this.line(this.point(x, y) + this.point(x, y+this.length));
    }
    tr.attr({'stroke-dasharray': '-', 'stroke' : color, 'stroke-width' : 1});
}

Thread.prototype.line = function(lines) {
    return paper.path("M" + lines);
}

var mad = new Madras();
mad.init();
</script>
