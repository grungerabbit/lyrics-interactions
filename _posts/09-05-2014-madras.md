<script>
var paper = Raphael(0,50,1000,1000);

var Madras = function(originX, originY, widthX, widthY) {
    this.originX = originX;
    this.originY = originY;
    this.widthX = widthX;
    this.widthY = widthY;    
}

Madras.prototype.init = function(threads) {
    for (var j = 0; j<threads; j++) {
        if (j%2 === 0) {
            var test = new Thread("vertical", 200, 0, (j*3));
            //test.generateDashes("green");
            test.defaultDash("orange");
            
             var test = new Thread("vertical", 300, 200, (j*3));
            test.defaultDash("green");
            
            var test2 = new Thread("horizontal", 600, (j*3), 0);
            test2.defaultDash("red");
        } else {
            var test = new Thread("vertical", 600, 2, (j*3));
            //test.generateDashes("green");
            test.defaultDash("orange");
            
            var test2 = new Thread("horizontal", 600, (j*3), 2);
            test2.defaultDash("red");
        }
    }
}

var Thread = function(direction, length, originX, originY) {
    this.direction = direction;
    this.length = length;
    this.originX = originX;
    this.originY = originY;
    this.multiplier = 4;
}

Thread.prototype.point = function (x, y) {
    return x + "," + y + ",";
}

Thread.prototype.numberOfDashes = function () {
    return parseInt(this.length / 4);
}

Thread.prototype.defaultDash = function (color) {
    var x = this.originX;
    var y = this.originY;
    var tr; 
    if (this.direction === "vertical") {
        tr = this.line(this.point(x, y) + this.point(x+this.length, y));
    }
    
    if (this.direction === "horizontal") {
        tr = this.line(this.point(x, y) + this.point(x, y+this.length));
    }
    tr.attr({'stroke-dasharray': '-', 'stroke' : color, 'stroke-width' : 1});
}

Thread.prototype.generateDashes = function (color) {
    var num = this.numberOfDashes();
    var m = this.multiplier;
    var x = this.originX;
    var y = this.originY;
    
    for (var i = 0; i<num; i++) {
        if (this.direction === "vertical") {
            var thr = this.line(this.point(m * i + x, y) + this.point(m * i+(m-1 + x), y));
            thr.attr({'stroke' : color});
        }
        
        if (this.direction === "horizontal") {
            console.log("horizontal! " + i);
            console.log(this.point(i, i));
        }
    }    
}

Thread.prototype.line = function(lines) {
    return paper.path("M" + lines);
}


var mad = new Madras();
mad.init(1000);
</script>
